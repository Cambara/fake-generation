    createFakeName(sex)
    {
        let firstName = this._createFirstName(sex);
        let surname = this._createSurname();
        return `${firstName} ${surname}`;
    }
 
    _createFirstName(sex)
    {
        let numberFName = this.mathService.getRandomNumber(0, this.names[`${sex}_names`].length);
        return this.names[`${sex}_names`][numberFName];
    }
 
    _createSurname()
    {
        let total = this.mathService.getRandomNumber(1,3);
        let surnames = [];
 
        for(let i = 0; i < total; i++)
        {
            let nSurname = this.mathService.getRandomNumber(0, this.names.surnames.length);
            surnames.push(this.names.surnames[nSurname]);
        }
        return surnames.join(" ");
    }
